int hourglassSum(vector<vector<int>> a) {
    int n=6,m=6;
    vector<int> hrg_sum;
    int ans=INT_MIN;
    for(int i=1;i<=n-2;i++)
    {
        for(int j=1;j<=n-2;j++)
        {
            int sum=0;
            sum=(a[i][j]+a[i-1][j]+a[i+1][j]);
            sum+=(a[i-1][j-1]+a[i-1][j+1]+a[i+1][j-1]+a[i+1][j+1]);
            hrg_sum.push_back(sum);
            if(ans<sum)
            ans=sum;
        }
    }
    return ans;
}
